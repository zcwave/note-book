void lineToNumberArray(vector<int> &result, string line) {
   int start = -1;
   for (int i = 0; i < line.length(); i++) {
      char ch = line[i];
      if (isdigit(ch)) {
         if (start == -1) start = i;
      } else {
         if (start >= 0) {
            result.push_back(atoi(line.substr(start, start - i).c_str()));
            start = -1;
         }
      }
   }
   if (start >= 0) result.push_back(atoi((line.substr(start).c_str())));
}

// 之后，通过floor(val, range), 将值进行向range取整，进而作为key访问map.
void render(int low, int high, int range, const vector<int> &rhs) {
   map<int, string> rank {};
   int l = low;

   while (l <= high) {
        rank[l] += integerToString(l) + "s: ";
        l += range;
    }

    for (auto i : rhs) {
        rank[floor(i, range)] += "*";
    }
    
    for (auto [key, val] : rank) {
        cout << val << endl;
    }
}
string integerToString(int value) {
    ostringstream oss;
    oss << value;
    // 暂不考虑失败？
    return oss.str();
}
int floor(int i, int range) {
    return int(i / range) * range;
}

#include <iostream>
#include <queue>
#include <vector>

using namespace std;

// 颠倒队列中的元素
void reverseQueue(queue<string> &queue) {
    vector<string> tempVector;

    // 将队列中的元素逐个压入向量中
    while (!queue.empty()) {
        tempVector.push_back(queue.front());
        queue.pop();
    }

    // 将向量中的元素逐个弹出并压回队列中（颠倒顺序）
    for (int i = tempVector.size() - 1; i >= 0; --i) {
        queue.push(tempVector[i]);
    }
}

int main() {
    queue<string> myQueue;

    // 向队列中添加一些元素
    myQueue.push("A");
    myQueue.push("B");
    myQueue.push("C");
    myQueue.push("D");

    // 打印原始队列中的元素（不改变队列内容）
    queue<string> originalQueue = myQueue; // 备份原始队列
    cout << "Original Queue:" << endl;
    while (!originalQueue.empty()) {
        cout << originalQueue.front() << " ";
        originalQueue.pop();
    }
    cout << endl;

    // 将队列中的元素颠倒
    reverseQueue(myQueue);

    // 打印颠倒后的队列中的元素
    cout << "Reversed Queue:" << endl;
    while (!myQueue.empty()) {
        cout << myQueue.front() << " ";
        myQueue.pop();
    }
    cout << endl;

    return 0;
}

# 习题四

编写一个程序显示一个表格。它根据单词的长度进行排序，并展示了不同长度的单词在英文字典中出现的次数。

```cpp
#include <iostream>
#include <map>

using namespace std;

int main() {
    // 定义一个map来存储单词长度及其出现的次数
    map<int, int> wordLengthFrequency;

    // 输入一些单词，以0结束输入
    string word;
    cout << "Enter words (enter 0 to stop):" << endl;
    while (true) {
        cin >> word;
        if (word == "0") {
            break;
        }
        wordLengthFrequency[word.length()]++;
    }

    // 输出表头
    cout << "Word Length\tFrequency" << endl;

    // 输出map中的内容，map已经按照键的大小排序
    for (const auto& pair : wordLengthFrequency) {
        cout << pair.first << "\t\t" << pair.second << endl;
    }

    return 0;
}
```
